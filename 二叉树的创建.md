#include<queue>

#include<iostream>
using namespace std;

template<class T>
struct BinTreeNode
{
	BinTreeNode<T> * _pLeft;
	BinTreeNode<T> * _pRight;
	T data;

	BinTreeNode(const T& data = 0 )
		:_pLeft(NULL)
		, _pRight(NULL)
		, data(data)
	{}
};

template<class T>
class BinaryTree
{
public:
	//构造函数
	BinaryTree()
		:_pRoot(NULL)
	{}

	//使用队列---层序建立二叉树
	bool CreateBinTree(T * arr,int sz)
	{
		if (arr == NULL || sz <= 0)
			return false;
		queue<BinTreeNode<T> * > s;//存放结构体指针的队列
		BinTreeNode<T> * temp;//当然队头的指针
		
		if (_pRoot == NULL)
		{
			_pRoot = new BinTreeNode<T>(arr[0]);
			s.push(_pRoot);
		}
		while (arr[0] != NULL)
		{
			
			
			
			int i = 1;
			while (i < sz)
			{
				printf("do\n");
				temp = s.front();
				if (temp->_pLeft == NULL)
				{
					temp->_pLeft = new BinTreeNode<T>(arr[i]);
					if (temp->_pLeft == NULL)
						return false;
					temp->_pLeft->data = arr[i];
					s.push(temp->_pLeft);
					i++;
					printf("左插入%d\n", temp->_pLeft->data);
				}
				else if (temp->_pRight == NULL)
				{
					temp->_pRight = new BinTreeNode<T>(arr[i]);
					if (temp->_pRight == NULL)
						return false;
					temp->_pRight->data = arr[i];
					s.push(temp->_pRight);
					i++;
					printf("右插入%d\n", temp->_pRight->data);
				}
				else
					s.pop();



			}
			return true;

		}
	}

	//前序遍历显示二叉树
	bool preOrder(BinTreeNode<T> * Node)
	{
		if (Node == NULL)
			return false;
		preOrder(Node->_pLeft);
		printf("%d", Node->data);
		preOrder(Node->_pRight);
	}


	BinTreeNode<T> * get()
	{
		BinTreeNode<T> * ret = _pRoot;
		return ret;
	}
	

private:
	BinTreeNode<T> * _pRoot;
};









int main()
{
	BinaryTree<int>  s;
	BinTreeNode<int> * pRoot;
	int a[8] = { 1, 2, 3, 4, 5, 6, 7, 8 };
	s.CreateBinTree(a,8);
	pRoot = s.get();
	s.preOrder(pRoot);
	return 0;
}
